Código de una lista 

Cualidades:

- Introduce información con ayuda de un botón
- Elimina informacion añadida con un botón

Cuerpo (pasos de su función)

Barra de información - checkbox (agregar)

se crea una barra y botón como consecuencia de oprimir
"agregar"

barra de información diasabled - botón (eliminar)


codigo:



<!DOCTYPE html>
<html lang="es">
<head>
  <meta charset="UTF-8">
  <title>Agregar Información</title>
  <style>
    body {
      font-family: Arial, sans-serif;
      padding: 10px;
    }
    .input-section {
      display: flex;
      gap: 10px;
      margin-bottom: 20px;
    }
    .info-list {
      display: flex;
      flex-direction: column;
      gap: 10px;
    }
    .info-item {
      display: flex;
      align-items: center;
      gap: 10px;
    }
    .info-bar {
      flex-grow: 1;
      padding: 5px 10px;
      background-color: #f0f0f0;
      border: 1px solid #ccc;
    }
    button {
      padding: 5px 10px;
      cursor: pointer;
    }
  </style>
</head>
<body>

  <h2>Agregar Información</h2>

  <div class="input-section">
    <input type="text" id="infoInput" placeholder="Escribe algo...">
    <button id="addBtn">Agregar</button>
  </div>

  <div class="info-list" id="infoList">
    </div>

  <script>
    // Obtener los elementos del DOM
    const addBtn = document.getElementById('addBtn');
    const infoInput = document.getElementById('infoInput');
    const infoList = document.getElementById('infoList');

    // ===================================================================
    // NUEVA FUNCIÓN PARA GUARDAR LA LISTA ACTUAL EN LOCALSTORAGE
    // ===================================================================
    function guardarTareas() {
      const tareasActuales = [];
      // Buscamos todos los elementos de la lista que están en la pantalla
      const elementosDeLista = document.querySelectorAll('.info-bar');
      // Por cada elemento, guardamos su texto en nuestro array
      elementosDeLista.forEach(function(elemento) {
        tareasActuales.push(elemento.textContent);
      });
      // Guardamos el array como un string en localStorage
      localStorage.setItem('misTareasGuardadas', JSON.stringify(tareasActuales));
    }

    // Función para agregar nueva información (TU CÓDIGO ORIGINAL CON 2 LÍNEAS NUEVAS)
    addBtn.addEventListener('click', () => {
      const texto = infoInput.value.trim();

      if (texto !== "") {
        // Crear contenedor del item
        const item = document.createElement('div');
        item.className = 'info-item';

        // Crear barra de información
        const infoBar = document.createElement('div');
        infoBar.className = 'info-bar';
        infoBar.textContent = texto;

        // Crear botón de eliminar
        const deleteBtn = document.createElement('button');
        deleteBtn.textContent = 'Eliminar';
        deleteBtn.addEventListener('click', () => {
          infoList.removeChild(item);
          guardarTareas(); // <-- 1. LÍNEA NUEVA: Guardamos al eliminar
        });

        // Añadir todo al contenedor
        item.appendChild(infoBar);
        item.appendChild(deleteBtn);
        infoList.appendChild(item);

        guardarTareas(); // <-- 2. LÍNEA NUEVA: Guardamos al agregar

        // Limpiar el input
        infoInput.value = "";
        infoInput.focus();
      }
    });

    // ===================================================================
    // NUEVO CÓDIGO QUE SE EJECUTA AL CARGAR LA PÁGINA
    // ===================================================================
    // Leemos las tareas que podrían estar guardadas
    const tareasGuardadas = localStorage.getItem('misTareasGuardadas');
    
    // Si había tareas guardadas...
    if (tareasGuardadas) {
      // Las convertimos de texto a un array
      const tareas = JSON.parse(tareasGuardadas);
      
      // Por cada una de las tareas guardadas...
      tareas.forEach(function(textoDeTarea) {
        // ...replicamos la lógica de creación de elementos que ya tenías
        const item = document.createElement('div');
        item.className = 'info-item';

        const infoBar = document.createElement('div');
        infoBar.className = 'info-bar';
        infoBar.textContent = textoDeTarea;

        const deleteBtn = document.createElement('button');
        deleteBtn.textContent = 'Eliminar';
        deleteBtn.addEventListener('click', () => {
          infoList.removeChild(item);
          guardarTareas(); // También aquí guardamos al eliminar
        });

        item.appendChild(infoBar);
        item.appendChild(deleteBtn);
        infoList.appendChild(item);
      });
    }
  </script>

</body>
</html>
